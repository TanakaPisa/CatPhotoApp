function largestOfFour(arr) {
  let big_sub_array=[];
  let compare = 0;
  for(let i in arr){
    for(let j = 0;j < arr[i].length; j++){
        if(arr[i][j]>arr[i][compare]){
          compare = j; 
        }
    }
    big_sub_array.push(arr[i][compare]);
    compare = 0;
  }
  return big_sub_array;
}

console.log(largestOfFour([[4, 5, 1, 3], [13, 27, 18, 26], [32, 35, 37, 39], [1000, 1001, 857, 1]]));
