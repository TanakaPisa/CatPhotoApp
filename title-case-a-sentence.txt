function titleCase(str) {
  let new_str = str.toLowerCase();
  let arr = new_str.split(" ")
  let upper_first = ""
  for(let i = 0; i < arr.length;i++){
    arr[i] = arr[i].replace(arr[i][0], arr[i][0].toUpperCase());
    if(i === 0){upper_first = arr[i]}
    else{upper_first += (" " +arr[i]);}
  }
  return upper_first;
}
